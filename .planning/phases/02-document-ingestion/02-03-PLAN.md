**src/app/api/documents/upload-url/route.ts:**

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { getUser } from '@/lib/supabase/server'
import { createAdminClient } from '@/lib/supabase/admin'
import { checkSubscription, isSubscriptionActive } from '@/lib/billing/subscription-check'
import { z } from 'zod'

const MAX_FILE_SIZE = 50 * 1024 * 1024 // 50MB

const uploadSchema = z.object({
  fileName: z.string().min(1),
  fileType: z.enum(['pdf', 'docx']),
  fileSize: z.number().positive().max(MAX_FILE_SIZE, 'File size exceeds 50MB limit'),
  title: z.string().min(1).max(200).optional(),
})

export async function POST(request: NextRequest) {
  // 1. Auth check
  const user = await getUser()
  if (!user) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  // 2. Subscription gate — per project convention, all new proposal creation gated
  const subscription = await checkSubscription(user.id)
  if (!isSubscriptionActive(subscription.status)) {
    return NextResponse.json(
      { error: 'Active subscription required to upload documents' },
      { status: 403 }
    )
  }

  // 3. Parse and validate request body
  const body = await request.json()
  const parsed = uploadSchema.safeParse(body)
  if (!parsed.success) {
    // Zod v4: use .issues not .errors
    return NextResponse.json(
      { error: 'Invalid request', issues: parsed.error.issues },
      { status: 400 }
    )
  }

  const { fileName, fileType, fileSize, title } = parsed.data
  const admin = createAdminClient()

  // 4. Create proposal row
  const { data: proposal, error: proposalError } = await admin
    .from('proposals')
    .insert({
      user_id: user.id,
      title: title || fileName.replace(/\.(pdf|docx)$/i, ''),
      status: 'processing',
      file_name: fileName,
      file_type: fileType,
    })
    .select('id')
    .single()

  if (proposalError || !proposal) {
    console.error('Failed to create proposal:', proposalError)
    return NextResponse.json({ error: 'Failed to create proposal' }, { status: 500 })
  }

  // 5. Build storage path: rfp-documents/{userId}/{proposalId}/original.{ext}
  const ext = fileType === 'pdf' ? 'pdf' : 'docx'
  const storagePath = `${user.id}/${proposal.id}/original.${ext}`

  // 6. Update proposal with storage path
  await admin
    .from('proposals')
    .update({ storage_path: storagePath })
    .eq('id', proposal.id)

  // 7. Create signed upload URL for direct browser upload
  const { data: signedUrl, error: signedUrlError } = await admin.storage
    .from('rfp-documents')
    .createSignedUploadUrl(storagePath)

  if (signedUrlError || !signedUrl) {
    console.error('Failed to create signed URL:', signedUrlError)
    // Clean up the proposal row
    await admin.from('proposals').delete().eq('id', proposal.id)
    return NextResponse.json({ error: 'Failed to create upload URL' }, { status: 500 })
  }

  // 8. Insert document_jobs row (status: pending — Edge Function picks it up)
  const { error: jobError } = await admin
    .from('document_jobs')
    .insert({
      proposal_id: proposal.id,
      user_id: user.id,
      storage_path: storagePath,
      file_type: fileType,
      status: 'pending',
    })

  if (jobError) {
    console.error('Failed to create job:', jobError)
    return NextResponse.json({ error: 'Failed to queue document for processing' }, { status: 500 })
  }

  // 9. Return signed URL + proposal ID to client
  return NextResponse.json({
    signedUrl: signedUrl.signedUrl,
    token: signedUrl.token,
    path: signedUrl.path,
    proposalId: proposal.id,
  })
}
```

Key implementation notes:
- Uses `createAdminClient()` (service_role) because the user-scoped client cannot create signed upload URLs for private buckets or insert rows for other tables with restrictive RLS.
- The signed upload URL is for the BROWSER to PUT the file directly to Supabase Storage (bypasses Vercel 4.5MB body limit).
- The `document_jobs` row starts as `pending`. The pg_cron-triggered Edge Function picks it up.
- On failure creating the signed URL, clean up the proposal row to avoid orphans.
- Zod v4: `parsed.error.issues` (not `.errors`).

**CRITICAL:** Supabase Edge Functions run in **Deno**, not Node.js. They CANNOT import from `@/lib/...` or use Node.js `require()`. The Edge Function must be self-contained or use Deno-compatible imports.

Since the parsing libraries (parse-pdf, parse-docx, textract, rfp-structure) are written for Node.js, the Edge Function must either:
- (a) Import them via npm specifiers (`npm:unpdf`, `npm:mammoth`) which Deno supports, OR
- (b) Contain inline implementations

Use approach (a) — npm specifiers in Deno Edge Functions.

**supabase/functions/process-documents/index.ts:**

```typescript
// Supabase Edge Function — runs on Deno runtime
// Triggered by pg_cron or HTTP invoke; claims and processes pending document jobs

import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'

// For PDF parsing in Deno Edge Function, we use the HTTP API approach:
// The Edge Function downloads the file, then processes inline.
// unpdf and mammoth need Node.js — so we use a simpler approach for the Edge Function.

// NOTE: If the Supabase Edge Function runtime supports npm: specifiers,
// use: import { extractText } from 'npm:unpdf/serverless'
// Otherwise, fall back to a fetch-based architecture where a Next.js API
// route does the parsing. Check Supabase Edge Function docs for npm support.

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
}

interface DocumentJob {
  id: string
  proposal_id: string
  user_id: string
  storage_path: string
  file_type: 'pdf' | 'docx'
  status: string
}

Deno.serve(async (req: Request) => {
  // Handle CORS preflight
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders })
  }

  const supabaseUrl = Deno.env.get('SUPABASE_URL')!
  const serviceRoleKey = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
  const awsAccessKey = Deno.env.get('AWS_ACCESS_KEY_ID')
  const awsSecretKey = Deno.env.get('AWS_SECRET_ACCESS_KEY')
  const awsRegion = Deno.env.get('AWS_REGION') || 'us-east-1'

  // Use service role — this function operates on behalf of the system
  const supabase = createClient(supabaseUrl, serviceRoleKey)

  try {
    // 1. Claim next pending job (atomic — prevents double processing)
    const { data: jobs, error: claimError } = await supabase
      .rpc('claim_next_document_job')

    if (claimError) {
      console.error('Failed to claim job:', claimError)
      return new Response(JSON.stringify({ error: 'Failed to claim job' }), {
        status: 500,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
      })
    }

    // rpc returns an array; claim function returns at most 1 row
    const job: DocumentJob | undefined = jobs?.[0]

    if (!job) {
      // No pending jobs — normal, just return
      return new Response(JSON.stringify({ message: 'No pending jobs' }), {
        status: 200,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
      })
    }

    console.log(`Processing job ${job.id} for proposal ${job.proposal_id}`)

    // 2. Download file from Supabase Storage
    const { data: fileData, error: downloadError } = await supabase.storage
      .from('rfp-documents')
      .download(job.storage_path)

    if (downloadError || !fileData) {
      await failJob(supabase, job, `Failed to download file: ${downloadError?.message}`)
      return new Response(JSON.stringify({ error: 'Download failed' }), {
        status: 500,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
      })
    }

    const arrayBuffer = await fileData.arrayBuffer()
    const uint8 = new Uint8Array(arrayBuffer)

    // 3. Parse text based on file type
    let text = ''
    let pageCount = 0
    let isScanned = false
    let ocrUsed = false

    if (job.file_type === 'pdf') {
      // PDF parsing using unpdf via npm specifier
      const { extractText } = await import('npm:unpdf/serverless')
      const result = await extractText(uint8, { mergePages: false })
      const pages: string[] = result.pages ?? [result.text ?? '']
      const charPerPage = pages.map((p: string) => p.length)
      text = pages.join('\n\n')
      pageCount = pages.length

      // Scanned PDF detection: all pages with < 100 chars
      isScanned = charPerPage.length > 0 && charPerPage.every((c: number) => c < 100)

      if (isScanned) {
        console.log(`Job ${job.id}: Detected scanned PDF, routing to OCR`)

        // Check size limit for Textract sync API (10MB)
        if (uint8.byteLength > 10_485_760) {
          await failJob(supabase, job,
            'Scanned PDF exceeds 10MB limit for OCR processing. Please upload a smaller file or a text-based PDF.')
          return new Response(JSON.stringify({ error: 'File too large for OCR' }), {
            status: 200,
            headers: { ...corsHeaders, 'Content-Type': 'application/json' },
          })
        }

        // Call AWS Textract DetectDocumentText (sync API)
        if (!awsAccessKey || !awsSecretKey) {
          await failJob(supabase, job, 'OCR not configured. AWS credentials missing.')
          return new Response(JSON.stringify({ error: 'OCR not configured' }), {
            status: 200,
            headers: { ...corsHeaders, 'Content-Type': 'application/json' },
          })
        }

        text = await callTextract(uint8, awsAccessKey, awsSecretKey, awsRegion)
        ocrUsed = true
      }
    } else if (job.file_type === 'docx') {
      // DOCX parsing using mammoth via npm specifier
      const mammoth = await import('npm:mammoth')
      const buffer = Buffer.from(arrayBuffer)
      const result = await mammoth.default.extractRawText({ buffer })
      text = result.value
      pageCount = 1 // DOCX doesn't have page concept in extraction
    }

    // 4. Extract RFP structure (regex heuristic — no AI)
    const rfpStructure = extractRfpStructure(text)

    // 5. Update proposals table with results
    const { error: updateError } = await supabase
      .from('proposals')
      .update({
        rfp_text: text,
        rfp_structure: rfpStructure,
        page_count: pageCount,
        is_scanned: isScanned,
        ocr_used: ocrUsed,
        status: 'ready',
        updated_at: new Date().toISOString(),
      })
      .eq('id', job.proposal_id)

    if (updateError) {
      await failJob(supabase, job, `Failed to update proposal: ${updateError.message}`)
      return new Response(JSON.stringify({ error: 'Update failed' }), {
        status: 500,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
      })
    }

    // 6. Mark job as completed
    await supabase
      .from('document_jobs')
      .update({
        status: 'completed',
        completed_at: new Date().toISOString(),
        updated_at: new Date().toISOString(),
      })
      .eq('id', job.id)

    console.log(`Job ${job.id} completed successfully`)

    return new Response(JSON.stringify({ success: true, jobId: job.id }), {
      status: 200,
      headers: { ...corsHeaders, 'Content-Type': 'application/json' },
    })

  } catch (err) {
    console.error('Unexpected error:', err)
    return new Response(JSON.stringify({ error: 'Internal error' }), {
      status: 500,
      headers: { ...corsHeaders, 'Content-Type': 'application/json' },
    })
  }
})

// --- Helper: Mark job as failed ---
async function failJob(
  supabase: ReturnType<typeof createClient>,
  job: DocumentJob,
  errorMessage: string
) {
  console.error(`Job ${job.id} failed: ${errorMessage}`)

  await supabase
    .from('document_jobs')
    .update({
      status: 'failed',
      error_message: errorMessage,
      completed_at: new Date().toISOString(),
      updated_at: new Date().toISOString(),
    })
    .eq('id', job.id)

  // Also update proposal status to indicate failure
  await supabase
    .from('proposals')
    .update({
      status: 'draft', // Reset to draft so user can retry
      updated_at: new Date().toISOString(),
    })
    .eq('id', job.proposal_id)
}

// --- Helper: AWS Textract sync call via fetch (no SDK needed in Deno) ---
async function callTextract(
  fileBytes: Uint8Array,
  accessKey: string,
  secretKey: string,
  region: string
): Promise<string> {
  // Use AWS Signature V4 for Textract API call
  // For Deno Edge Functions, we use the fetch API with AWS signing
  // rather than the full @aws-sdk which is Node.js-centric

  const endpoint = `https://textract.${region}.amazonaws.com`
  const body = JSON.stringify({
    Document: {
      Bytes: btoa(String.fromCharCode(...fileBytes)),
    },
  })

  // AWS Signature V4 signing
  const now = new Date()
  const dateStamp = now.toISOString().replace(/[-:]/g, '').split('.')[0] + 'Z'
  const shortDate = dateStamp.slice(0, 8)

  const encoder = new TextEncoder()

  async function hmacSha256(key: ArrayBuffer, message: string): Promise<ArrayBuffer> {
    const cryptoKey = await crypto.subtle.importKey(
      'raw', key, { name: 'HMAC', hash: 'SHA-256' }, false, ['sign']
    )
    return crypto.subtle.sign('HMAC', cryptoKey, encoder.encode(message))
  }

  async function sha256(message: string): Promise<string> {
    const hash = await crypto.subtle.digest('SHA-256', encoder.encode(message))
    return Array.from(new Uint8Array(hash)).map(b => b.toString(16).padStart(2, '0')).join('')
  }

  const payloadHash = await sha256(body)
  const canonicalHeaders = [
    `content-type:application/x-amz-json-1.1`,
    `host:textract.${region}.amazonaws.com`,
    `x-amz-date:${dateStamp}`,
    `x-amz-target:Textract.DetectDocumentText`,
  ].join('\n') + '\n'

  const signedHeaders = 'content-type;host;x-amz-date;x-amz-target'
  const canonicalRequest = [
    'POST', '/', '', canonicalHeaders, signedHeaders, payloadHash,
  ].join('\n')

  const credentialScope = `${shortDate}/${region}/textract/aws4_request`
  const stringToSign = [
    'AWS4-HMAC-SHA256', dateStamp, credentialScope, await sha256(canonicalRequest),
  ].join('\n')

  // Derive signing key
  let signingKey = encoder.encode(`AWS4${secretKey}`).buffer as ArrayBuffer
  for (const part of [shortDate, region, 'textract', 'aws4_request']) {
    signingKey = await hmacSha256(signingKey, part)
  }
  const signatureBytes = await hmacSha256(signingKey, stringToSign)
  const signature = Array.from(new Uint8Array(signatureBytes))
    .map(b => b.toString(16).padStart(2, '0')).join('')

  const authorization = `AWS4-HMAC-SHA256 Credential=${accessKey}/${credentialScope}, SignedHeaders=${signedHeaders}, Signature=${signature}`

  const response = await fetch(endpoint, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/x-amz-json-1.1',
      'X-Amz-Date': dateStamp,
      'X-Amz-Target': 'Textract.DetectDocumentText',
      'Authorization': authorization,
    },
    body,
  })

  if (!response.ok) {
    const errorText = await response.text()
    throw new Error(`Textract API error (${response.status}): ${errorText}`)
  }

  const result = await response.json()
  const lines = (result.Blocks ?? [])
    .filter((block: { BlockType: string }) => block.BlockType === 'LINE')
    .map((block: { Text?: string }) => block.Text ?? '')

  return lines.join('\n')
}

// --- Inline: RFP structure extraction (same logic as src/lib/documents/rfp-structure.ts) ---
// Duplicated here because Edge Functions (Deno) cannot import from src/lib/
interface RfpSection {
  number: string
  title: string
}

interface RfpRequirement {
  text: string
  type: 'shall' | 'must' | 'will' | 'should'
  sectionRef?: string
}

interface RfpStructure {
  sections: RfpSection[]
  requirements: RfpRequirement[]
}

function extractRfpStructure(text: string): RfpStructure {
  const sections: RfpSection[] = []
  const requirements: RfpRequirement[] = []

  // Section patterns
  const sectionRegex = /^(?:(?:SECTION\s+)?([A-Z](?:\.\d+)*)\s*[-.:]\s*(.+)|(\d+(?:\.\d+)*)\s+([A-Z][^\n]{2,}))/gm
  let match: RegExpExecArray | null
  while ((match = sectionRegex.exec(text)) !== null) {
    const number = (match[1] || match[3] || '').trim()
    const title = (match[2] || match[4] || '').trim()
    if (number && title && title.length < 200) {
      sections.push({ number, title })
    }
  }

  // Requirement extraction
  const sentences = text.split(/(?<=[.!?])\s+|(?<=\n)\s*/)
  const kwRegex = /\b(shall|must|will|should)\b/i
  let currentSection: string | undefined

  for (const sentence of sentences) {
    const secMatch = sentence.match(/^(?:(?:SECTION\s+)?([A-Z](?:\.\d+)*)\s*[-.:])|(^\d+(?:\.\d+)*)\s/)
    if (secMatch) {
      currentSection = (secMatch[1] || secMatch[2] || '').trim()
    }

    const kwMatch = sentence.match(kwRegex)
    if (kwMatch && sentence.length > 20 && sentence.length < 1000) {
      requirements.push({
        text: sentence.trim(),
        type: kwMatch[1].toLowerCase() as RfpRequirement['type'],
        sectionRef: currentSection,
      })
    }
  }

  return { sections, requirements }
}
```

**IMPORTANT implementation notes for the executor:**

1. **Deno npm specifiers:** The Edge Function uses `await import('npm:unpdf/serverless')` and `await import('npm:mammoth')`. If Supabase Edge Functions don't support dynamic npm imports, switch to static imports at the top: `import { extractText } from 'npm:unpdf/serverless'` and `import mammoth from 'npm:mammoth'`. Check Supabase docs.

2. **AWS Textract in Deno:** Instead of using `@aws-sdk/client-textract` (Node.js SDK), the Edge Function uses raw `fetch()` with AWS Signature V4 signing. This avoids Node.js dependency. The `callTextract()` helper implements minimal SigV4.

3. **Base64 encoding for Textract:** The Textract API expects `Document.Bytes` as a base64-encoded string in JSON. The `btoa()` approach works for files up to ~10MB. For very large Uint8Arrays, chunked encoding may be needed — but since we reject > 10MB before calling Textract, this is safe.

4. **RFP structure is duplicated inline** because Deno Edge Functions cannot import from `src/lib/`. This is intentional. Keep both in sync if the heuristic changes.

5. **Error handling:** Any exception during processing calls `failJob()` which sets `document_jobs.status = 'failed'` with the error message and resets `proposals.status = 'draft'` so the user can retry.

6. **mammoth in Deno:** If `npm:mammoth` doesn't work in Edge Functions, an alternative is to only support PDF in the Edge Function and handle DOCX via a separate Next.js API route (DOCX extraction is lightweight enough for Vercel). Flag this if it occurs.
