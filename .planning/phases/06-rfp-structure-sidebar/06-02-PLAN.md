    **Add click-to-scroll handler:**
    ```typescript
    const handleRfpSectionClick = useCallback((sectionTitle: string) => {
      const editor = editorRef.current?.editor
      if (!editor) return

      // Find the first heading node whose text matches the section title
      let targetPos: number | null = null
      editor.state.doc.descendants((node, pos) => {
        if (targetPos !== null) return false // stop after first match
        if (node.type.name === 'heading' && node.textContent.toLowerCase().includes(sectionTitle.toLowerCase())) {
          targetPos = pos
          return false
        }
      })

      if (targetPos !== null) {
        editor.commands.setTextSelection(targetPos)
        editor.commands.scrollIntoView()
        editor.commands.focus()
      }
    }, [])
    ```

    **Add active section detection via editor selection/update:**

    Create a helper function (outside the component or inside with useCallback) that finds the nearest heading above the current cursor position:
    ```typescript
    const detectActiveRfpSection = useCallback(() => {
      const editor = editorRef.current?.editor
      if (!editor || !rfpStructure) return

      const { from } = editor.state.selection
      let nearestHeading: string | null = null

      // Walk backwards from cursor to find the closest heading above
      editor.state.doc.nodesBetween(0, from, (node) => {
        if (node.type.name === 'heading') {
          const headingText = node.textContent
          // Match against rfp_structure section titles
          const matchedSection = rfpStructure.sections.find(
            (s) => headingText.toLowerCase().includes(s.title.toLowerCase())
          )
          if (matchedSection) {
            nearestHeading = matchedSection.number
          }
        }
      })

      setActiveRfpSection(nearestHeading)
    }, [rfpStructure])
    ```

    **Wire detection to editor events:**

    Add a `useEffect` that listens to the Tiptap editor's `selectionUpdate` and `update` events:
    ```typescript
    useEffect(() => {
      const editor = editorRef.current?.editor
      if (!editor) return

      const handler = () => detectActiveRfpSection()
      editor.on('selectionUpdate', handler)
      editor.on('update', handler)

      return () => {
        editor.off('selectionUpdate', handler)
        editor.off('update', handler)
      }
    }, [detectActiveRfpSection])
    ```

    Note: The editor ref may not be available on first render. Add a second `useEffect` that runs `detectActiveRfpSection()` once on mount with a short delay:
    ```typescript
    useEffect(() => {
      const timer = setTimeout(() => detectActiveRfpSection(), 500)
      return () => clearTimeout(timer)
    }, [detectActiveRfpSection, activeSection]) // re-run when tab changes
    ```
